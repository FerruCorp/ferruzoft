Este ejemplo mostrará en consola que se ha creado un objeto con una propiedad llamada "propl". El código es el siguiente:
var obj = {};
obj.propl = 3;
console.log(obj);

Este ejemplo mostrará en consola que se ha creado una funcion la cual como objeto tiene una propiedad nombrada como "myProp". El código es el siguiente:
function log(arg){
console.log(arg);
}
log.myProp = 2;
console.log(log);
console.dir(log);

Este ejemplo mostrará en consola que se ha creado un arreglo de 3 elementos y aparte tiene una propiedad "myProp" con un valor de 3. El código es el siguiente:
var ar = [3];
ar.myProp = 3;
console.log(ar);

Este ejemplo mostrará en consola que se ha creado un objeto con una propiedad "name" y con valor "Joe". Ademas, se le agregará propiedades por medio de un bucle for. El código es el siguiente:
var obj = {};
obj.name = 'Joe';
console.log(obj.name);
console.log(obj['name']);
for(var i=0; i<5; i++){
obj['prop' + i] = i;
}
console.log(obj);
